Real-Time Chatbot with Streaming Python
import asyncio import time from typing import AsyncGenerator from transformers import AutoTokenizer, AutoModelForCausalLM import torch class StreamingChatbot: def __init__(self, model_name="microsoft/DialoGPT-medium"): # Load conversational model self.tokenizer = AutoTokenizer.from_pretrained(model_name) self.model = AutoModelForCausalLM.from_pretrained(model_name) self.chat_history_ids = None print(f"✓ Loaded {model_name}") async def stream_response(self, user_input: str) -> AsyncGenerator[str, None]: """Stream tokens as they're generated""" start_time = time.time() # Encode input new_input_ids = self.tokenizer.encode( user_input + self.tokenizer.eos_token, return_tensors='pt' ) # Append to chat history if self.chat_history_ids is not None: bot_input_ids = torch.cat([self.chat_history_ids, new_input_ids], dim=-1) else: bot_input_ids = new_input_ids # Generate with streaming generated_tokens = [] for i in range(100): # Max tokens outputs = self.model(bot_input_ids) next_token_logits = outputs.logits[:, -1, :] next_token = torch.argmax(next_token_logits, dim=-1).unsqueeze(-1) # Check for end token if next_token.item() == self.tokenizer.eos_token_id: break # Decode and yield token token_str = self.tokenizer.decode(next_token[0], skip_special_tokens=True) yield token_str # Update context bot_input_ids = torch.cat([bot_input_ids, next_token], dim=-1) generated_tokens.append(next_token.item()) # Small delay to simulate streaming await asyncio.sleep(0.01) # Save chat history self.chat_history_ids = bot_input_ids latency = (time.time() - start_time) * 1000 yield f"\n[Latency: {latency:.0f}ms]" # Example usage async def demo_streaming_chat(): bot = StreamingChatbot() conversations = [ "Hello! How are you today?", "I need help with my order.", "Can you check my shipping status?" ] for user_msg in conversations: print(f"\n👤 User: {user_msg}") print("🤖 Bot: ", end="") async for token in bot.stream_response(user_msg): print(token, end="", flush=True) print() # Run demo await demo_streaming_chat()

Voice Integration with Streaming ASR/TTS

Voice Assistant Pipeline Python
import speech_recognition as sr from gtts import gTTS import tempfile import os class VoiceAssistant: def __init__(self): self.recognizer = sr.Recognizer() self.chatbot = StreamingChatbot() def listen(self) -> str: """Capture and transcribe speech""" with sr.Microphone() as source: print("🎤 Listening...") self.recognizer.adjust_for_ambient_noise(source, duration=0.5) start_time = time.time() audio = self.recognizer.listen(source, timeout=5) # Transcribe (ASR) try: text = self.recognizer.recognize_google(audio) asr_latency = (time.time() - start_time) * 1000 print(f"📝 Transcribed: '{text}' ({asr_latency:.0f}ms)") return text except sr.UnknownValueError: return "" def speak(self, text: str): """Convert text to speech""" start_time = time.time() # Generate speech tts = gTTS(text=text, lang='en') with tempfile.NamedTemporaryFile(delete=False, suffix='.mp3') as fp: tts.save(fp.name) tts_latency = (time.time() - start_time) * 1000 print(f"🔊 TTS Generated ({tts_latency:.0f}ms)") # Play audio (platform-specific) os.system(f"mpg123 {fp.name}") # Linux # os.system(f"afplay {fp.name}") # macOS os.unlink(fp.name) async def conversation_loop(self): """Full voice conversation""" print("🎙️ Voice Assistant Ready!") while True: # Listen user_input = self.listen() if not user_input: continue if "goodbye" in user_input.lower(): self.speak("Goodbye! Have a great day!") break # Generate response response = "" async for token in self.chatbot.stream_response(user_input): if "Latency" not in token: response += token # Speak response print(f"🤖 Response: {response}") self.speak(response)

⚡ Latency Breakdown

Component Latency Optimization
ASR (Speech → Text) 100-500ms Streaming ASR, VAD
NLU (Intent Detection) 10-50ms Cached models, distillation
Dialogue Manager 50-200ms Rule-based fallbacks
Response Generation 200-1000ms Streaming, KV cache
TTS (Text → Speech) 100-400ms Neural TTS, caching

✅ Best Practices

  • Streaming: Start TTS before full response is generated
  • Turn Detection: Use VAD to detect when user stops speaking
  • Context Management: Limit context window to last 3-5 turns
  • Fallbacks: Have rule-based responses for common intents
  • Caching: Cache frequent responses and TTS audio

Real-Time Speech Recognition

Streaming ASR with Whisper Python
import whisper import numpy as np import pyaudio from collections import deque import threading class StreamingASR: def __init__(self, model_size="base"): # Load Whisper model self.model = whisper.load_model(model_size) print(f"✓ Loaded Whisper {model_size}") # Audio config self.CHUNK = 1024 self.FORMAT = pyaudio.paFloat32 self.CHANNELS = 1 self.RATE = 16000 # Buffer for audio chunks self.audio_buffer = deque(maxlen=50) # ~3 seconds self.is_listening = False def audio_callback(self, in_data, frame_count, time_info, status): """Process audio chunks""" audio_data = np.frombuffer(in_data, dtype=np.float32) self.audio_buffer.append(audio_data) return (in_data, pyaudio.paContinue) def start_listening(self): """Start audio stream""" self.is_listening = True p = pyaudio.PyAudio() stream = p.open( format=self.FORMAT, channels=self.CHANNELS, rate=self.RATE, input=True, frames_per_buffer=self.CHUNK, stream_callback=self.audio_callback ) stream.start_stream() print("🎤 Listening started...") return stream, p def transcribe_buffer(self): """Transcribe accumulated audio""" if len(self.audio_buffer) < 10: return "" start_time = time.time() # Concatenate buffer audio_data = np.concatenate(list(self.audio_buffer)) # Transcribe result = self.model.transcribe(audio_data, fp16=False) latency = (time.time() - start_time) * 1000 return { 'text': result['text'], 'language': result.get('language', 'en'), 'latency_ms': latency } def live_transcription(self, duration=10): """Run live transcription for N seconds""" stream, p = self.start_listening() try: for i in range(duration // 2): time.sleep(2) # Transcribe every 2 seconds result = self.transcribe_buffer() if result: print(f"\n[{time.strftime('%H:%M:%S')}] " f"({result['language']}) " f"{result['text']} " f"[{result['latency_ms']:.0f}ms]") finally: stream.stop_stream() stream.close() p.terminate() # Demo asr = StreamingASR(model_size="base") asr.live_transcription(duration=10)

Real-Time Object Detection on Video

Live CCTV Analytics Python
import cv2 from ultralytics import YOLO import time from collections import deque class VideoAnalytics: def __init__(self, model_name="yolov8n.pt"): self.model = YOLO(model_name) self.fps_buffer = deque(maxlen=30) # Analytics trackers self.person_count = 0 self.alerts = [] print(f"✓ Video Analytics initialized") def process_frame(self, frame): """Process single frame""" start_time = time.time() # Run detection results = self.model(frame, verbose=False) detections = results[0] # Count people people = [d for d in detections.boxes if int(d.cls[0]) == 0] # person class # Draw boxes annotated = detections.plot() # Calculate FPS latency = (time.time() - start_time) * 1000 fps = 1000 / latency if latency > 0 else 0 self.fps_buffer.append(fps) # Add metrics overlay avg_fps = sum(self.fps_buffer) / len(self.fps_buffer) cv2.putText(annotated, f"FPS: {avg_fps:.1f}", (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0), 2) cv2.putText(annotated, f"People: {len(people)}", (10, 70), cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0), 2) cv2.putText(annotated, f"Latency: {latency:.0f}ms", (10, 110), cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0), 2) return annotated, len(people), latency def analyze_stream(self, video_source=0, duration=10): """Analyze video stream""" cap = cv2.VideoCapture(video_source) print(f"📹 Processing video stream...") frame_count = 0 total_latency = 0 max_people = 0 start_time = time.time() while time.time() - start_time < duration: ret, frame = cap.read() if not ret: break # Process frame annotated, people_count, latency = self.process_frame(frame) # Track metrics frame_count += 1 total_latency += latency max_people = max(max_people, people_count) # Display (optional) # cv2.imshow('Video Analytics', annotated) if frame_count % 30 == 0: print(f"Frame {frame_count}: " f"{people_count} people, " f"{latency:.0f}ms latency") cap.release() # Summary avg_latency = total_latency / frame_count avg_fps = frame_count / duration print(f"\n📊 Summary:") print(f" Frames processed: {frame_count}") print(f" Average FPS: {avg_fps:.1f}") print(f" Average latency: {avg_latency:.0f}ms") print(f" Max people detected: {max_people}") # Demo analytics = VideoAnalytics() analytics.analyze_stream(video_source="test_video.mp4", duration=10)

✅ Optimization Strategies

  • Frame Skipping: Process every Nth frame for non-critical applications
  • Resolution Scaling: Downscale frames to 640x480 or 416x416
  • Model Choice: YOLOv8n (nano) for speed, YOLOv8x for accuracy
  • GPU Batching: Process multiple frames in parallel
  • ROI Processing: Only analyze regions of interest
